) ENGINE = MergeTree()

ORDER BY (user_id, timestamp)
PRIMARY KEY user_id
SAMPLE BY user_id
TTL timestamp + INTERVAL 30 DAY
SETTINGS index_granularity = 8192
COMMENT 'Event tracking table';
